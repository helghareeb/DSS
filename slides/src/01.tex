\begin{frame}{Learning objectives}
  \begin{itemize}
    \item Define DSS and distinguish them from traditional information systems.
    \item Identify core DSS components and major DSS types.
    \item Explain how AI, ML, and DL extend classical DSS.
    \item Discuss human oversight, explainability, and ethics in modern DSS.
  \end{itemize}
\end{frame}

\begin{frame}{What is a DSS?}
  \begin{itemize}
    \item A \keyterm{Decision Support System} supports semi-structured and
    unstructured decisions.
    \item It combines \keyterm{data}, \keyterm{models}, \keyterm{knowledge},
    and \keyterm{interfaces}.
    \item The goal is to improve judgment, not always to replace the decision
    maker.
  \end{itemize}
\end{frame}

\begin{frame}{Classical DSS architecture}
  \begin{itemize}
    \item \keyterm{Data management}: databases, warehouses, external sources.
    \item \keyterm{Model management}: optimization, simulation, forecasting.
    \item \keyterm{Knowledge management}: rules and heuristics.
    \item \keyterm{Dialog subsystem}: dashboards, queries, visual analytics.
  \end{itemize}
\end{frame}

\begin{frame}{From classical DSS to AI-enabled DSS}
  \begin{itemize}
    \item ML adds prediction and classification.
    \item DL handles text, images, and other unstructured data.
    \item AI-enabled DSS need monitoring, explanation, and governance.
  \end{itemize}
\end{frame}

\begin{frame}{Human-centered DSS}
  \begin{itemize}
    \item Human-in-the-loop design remains critical in high-stakes contexts.
    \item Explainability improves trust and contestability.
    \item Fairness, privacy, and accountability must be designed explicitly.
  \end{itemize}
\end{frame}

\discussionframe{Choose one DSS domain such as healthcare, education, or
finance. What should remain under human judgment even if the system becomes
highly accurate?}

\begin{frame}{Key takeaways}
  \begin{itemize}
    \item DSS evolved from reporting and modeling platforms into AI-enabled
    decision environments.
    \item Better prediction does not remove the need for governance.
    \item Strong DSS design combines analytics quality with decision quality.
  \end{itemize}
\end{frame}
