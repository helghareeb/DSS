\begin{frame}{Learning objectives}
  \begin{itemize}
    \item Compare normative, descriptive, and prescriptive views.
    \item Explain bounded rationality and common decision biases.
    \item Distinguish decisions under certainty, risk, and uncertainty.
    \item Connect decision concepts to DSS design choices.
  \end{itemize}
\end{frame}

\begin{frame}{Three views of decision making}
  \begin{itemize}
    \item \keyterm{Normative}: how decisions should be made.
    \item \keyterm{Descriptive}: how people actually decide.
    \item \keyterm{Prescriptive}: how systems can improve decisions in
    practice.
  \end{itemize}
\end{frame}

\begin{frame}{Bounded rationality and heuristics}
  \begin{itemize}
    \item Decision makers rarely optimize perfectly.
    \item They satisfice under time, data, and cognitive limits.
    \item Common biases: anchoring, availability, confirmation, overconfidence,
    automation bias.
  \end{itemize}
\end{frame}

\begin{frame}{Decision environments}
  \begin{itemize}
    \item \keyterm{Certainty}: outcomes are known.
    \item \keyterm{Risk}: probabilities can be estimated.
    \item \keyterm{Uncertainty}: probabilities are weak or unavailable.
  \end{itemize}
\end{frame}

\begin{frame}{Tools that support decisions}
  \begin{itemize}
    \item Decision trees structure sequential choices.
    \item Expected value and expected utility quantify trade-offs.
    \item Value of information helps prioritize additional data collection.
  \end{itemize}
\end{frame}

\discussionframe{Which human bias is most dangerous in AI-enabled DSS:
automation bias, confirmation bias, or overconfidence? Defend your answer with
one example.}

\begin{frame}{Key takeaways}
  \begin{itemize}
    \item DSS must reflect how decisions are made, not just how they should be
    made.
    \item Bias mitigation and uncertainty representation are core DSS features.
  \end{itemize}
\end{frame}
