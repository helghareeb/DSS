\begin{frame}{Learning objectives}
  \begin{itemize}
    \item Relate BI to DSS.
    \item Explain warehousing, data acquisition, and semantic layers.
    \item Describe how data mining supports BI.
    \item Apply visualization principles for decision-focused dashboards.
  \end{itemize}
\end{frame}

\begin{frame}{BI as the foundation of DSS}
  \begin{itemize}
    \item BI makes performance observable through trusted metrics.
    \item DSS extends BI by recommending actions and evaluating options.
    \item Good BI is required for good decision support.
  \end{itemize}
\end{frame}

\begin{frame}{Warehousing and data acquisition}
  \begin{itemize}
    \item Warehouses integrate historical analytical data.
    \item Fact and dimension tables support intuitive analysis.
    \item ETL, ELT, CDC, and streaming shape timeliness.
  \end{itemize}
\end{frame}

\begin{frame}{From mining to action}
  \begin{itemize}
    \item Segmentation, association, and anomaly patterns matter.
    \item Patterns become useful when linked to interventions and playbooks.
    \item Decision ownership should be explicit.
  \end{itemize}
\end{frame}

\begin{frame}{Visualization for decisions}
  \begin{itemize}
    \item Match the chart to the question.
    \item Avoid misleading scales and overloaded dashboards.
    \item Add baselines, context, and next actions.
  \end{itemize}
\end{frame}

\discussionframe{What is the difference between a dashboard that informs and a
dashboard that truly supports decisions?}

\begin{frame}{Key takeaways}
  \begin{itemize}
    \item BI becomes DSS when information is connected to action.
    \item Consistent KPI definitions and data quality are non-negotiable.
  \end{itemize}
\end{frame}
