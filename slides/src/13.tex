\begin{frame}{Purpose of this chapter}
  \begin{itemize}
    \item Support exam preparation and assignment design.
    \item Show the expected depth of strong written answers.
    \item Help students connect concepts across multiple chapters.
  \end{itemize}
\end{frame}

\begin{frame}{Typical essay themes}
  \begin{itemize}
    \item MIS vs. BI vs. DSS.
    \item Human-in-the-loop design in AI-enabled DSS.
    \item Model management and governance.
    \item Supervised vs. unsupervised learning.
    \item Expert systems vs. ML systems.
  \end{itemize}
\end{frame}

\begin{frame}{What makes a strong essay answer?}
  \begin{itemize}
    \item Clear structure and correct terminology.
    \item Comparison, not only definition.
    \item One concrete example.
    \item Attention to governance, fairness, and implementation.
  \end{itemize}
\end{frame}

\begin{frame}{Common weaknesses}
  \begin{itemize}
    \item Giving tool lists instead of an argument.
    \item Ignoring trade-offs and assumptions.
    \item Treating model accuracy as the only success criterion.
    \item Forgetting the human and organizational context.
  \end{itemize}
\end{frame}

\discussionframe{If you were grading DSS essay answers, what is the single most
important sign that a student actually understands the subject?}

\begin{frame}{Revision advice}
  \begin{itemize}
    \item Build answers around the decision problem.
    \item Reuse key terms precisely.
    \item Connect analytics methods to organizational action.
  \end{itemize}
\end{frame}
